
\begin{scheme}[H]
    \caption{$\quad\fn{Additive \;Secret \;Sharing \;(SS)}$} \label{alg:additive}
    \begin{algorithmic}[1]
    \begin{alginfo}
        Additive Secret Sharing Scheme. This scheme is parametrized by a group $\G$ in  and the number of shares produced $n$. 
    \end{alginfo}
    
    \vspace{-10pt}
    \AlgFnct{Split}{n}{$x \in \G$}{$(\vect{y} \equals \{\vecti{y}{1}, \dots \vecti{y}{n}\})$}
        \For {$i \in \range{n-1}$}
            \State Sample $\sample{\vecti{y}{i}}{\G}$ as $n\shortminus 1$ random shares
        \EndFor
        \State $\assign{\vecti{y}{n}}{x - \sum 
        _{i ins \range{n-1}}\vecti{y}{i}}$ as the last random share
        \Statex \Return $\vect{y}\equals\{\vecti{y}{i}\}_{i \in \range{n}}$ as the $n$ shares

    \vspace{-10pt}
    \AlgFnct{\hspace{-6pt}Combine}{n}{$\vect{y'}\equals\{\vecti{y}{i} \ins \G\}_{i\in \range{n}}$}{$x$}
    \setalglineno{1}
        \State $\assign{x}{\sum_{i \in \range{n}}\vecti{y}{i}}$
        \Statex \Return $x$ as the reconstructed secret
    \end{algorithmic}
\end{scheme}
