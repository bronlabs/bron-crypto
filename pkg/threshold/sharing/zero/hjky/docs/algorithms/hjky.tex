\begin{protocol}[ht]
    \caption{$\quad\fndef{HJKY}$}
    \label{alg:hjky}
    \begin{algorithmic}[1]
        \begin{alginfo}
            An interactive protocol~\cite{herzberg1995proactive} based on \fnref{PedersenDKG} to generate shares of zero among $n$ parties, parametrized by a group $\G$ with identity $\g{I}$.
        \end{alginfo} 
        
        \Players { $\party{1}, \dots,\party{n}$, with symmetric behavior.}

        \Ensure {$\party{i}$: $o_i$, a share of zero $\forall i \inss \range{n}$ s.t. $\sum_{i \in \range{n}} o_i = 0_\G$}

        \vspace{-8pt}
        \AlgRoundZero{\hspace{-2pt}$\party{i}$}{\fndef[hjky]{Round1}}{}{$(\vect{x}_i, \vect{\g{C}}_i)$}
        \State Set $\sample{a_{i,0}}{0}$
        \State Run steps 2-3 of $\fnref{PedersenDKG}.\fnref[pedersendkg]{Round1}()$, obtaining $\vecti{x}{i,j}\;\;\forall j \ins \range{n}$ and $\vect{\g{C}}_i$
        \State $\sendTo{\party{j}}{\vecti{x}{i,j}}\;\;\forall j \ins \range{n}$
        \State $\idealfnref{Broadcast}(\vect{\g{C}}_i)$

        \vspace{-8pt}
        \setalglineno{1}
        \AlgRound{\hspace{-2pt}$\party{i}$}{\fndef[hjky]{Round2}}{$\{\vect{\g{C}}_j, \mathit{\pi}_j\}_{j\in\range{n}}$}{($o_i$)}
        \State Run $\assign{o_i, \g{Y}}{\fnref{PedersenDKG}.\fnref[pedersendkg]{Round2}(\{\vect{\g{C}}_j, \mathit{\pi}_j\}_{\forall j})}$
        \State Check if $\isequal{\vect{\g{C}}_{j,0}}{\g{I}}$; otherwise $\ABORT$
        \State Check if $\isequal{\g{Y}}{\g{I}}$; otherwise $\ABORT$ (sanity check)
        \Statex \Return $o_i$
    \end{algorithmic}
\end{protocol}
