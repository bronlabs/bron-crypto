\begin{functionality}
    \caption{$\quad\idealfndef{PRNG}$}
    \label{alg:prng}
    \begin{algorithmic}[1]
        \begin{alginfo}
            Pseudo-Random Number Generator, defined by two algorithms:
            \begin{itemize}[leftmargin=12pt]
                \item $\idealfnref{PRNG}.\fn{Seed}(s)$ seeds a PRNG instance with seed $s$.
                \item $\idealfnref{PRNG}.\fn{Sample}(n)\rightarrow u \in \{0,1\}^n$ generates $n$ uniformly random bits.
            \end{itemize}
            In cases where a prng needs to be seeded multiple times, we simplify to $\assign{x}{\fn{PRNG}_n(s)}$ to represent $\idealfnref{PRNG}.\fn{Seed}(s)$ followed by $\assign{x}{\idealfnref{PRNG}.\fn{Sample}}(n)$
        \end{alginfo}
    \end{algorithmic}
\end{functionality}
