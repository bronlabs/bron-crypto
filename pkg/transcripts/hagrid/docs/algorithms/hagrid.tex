\begin{scheme}[ht]
    \caption{$\quad$\fndef{Transcript} (\fndef{T})} \label{alg:transcript}
    \begin{algorithmic}[1]
        \begin{alginfo}
            A transcript abstraction holding a state $s \in \Z_2^\kappa$, a condensed description of the messages exchanged in a protocol execution. It is parametrized by a hash function \fn{H} (we employ SHA3-256~\cite{FIPS202}), a \fn{PRNG} (we use Shake256~\cite{FIPS202}). The state $s$ is initialized to zero and updated upon each \fn{Append} / \fn{Extract}.
        \end{alginfo}

        \vspace{-12pt}
        \AlgFnct{$\fn{T}.\fndef[t]{Append}_{l}$}{}{$m$}{}
        \State $\assign{\fn{T}.s}{\fn{H}(\fn{H}(\fn{T}.s \,||\,l) \,||\,m)}$ with message $m$ and label $l$ to update $s$
  
        \vspace{-12pt}
        \AlgFnct{\hspace{-5pt}$\fn{T}.\fndef[t]{Extract}_{l}$}{}{$k \ins \mathbb{N}$}{$r\ins\Z_2^k$}
        \State $\assign{\fn{T}.s}{\fn{H}(\fn{T}.s \,||\,l)}$ with label $l$ to update $s$
        \State Set $\fn{PRNG.Seed}(\fn{T}.s)$ and compute $\assign{r}{\fn{PRNG.Get}(k)}$ to get the randomness.
        \Statex \Return $r$
        
    \end{algorithmic}
\end{scheme}
