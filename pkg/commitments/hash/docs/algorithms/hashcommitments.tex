\begin{scheme}[H]
    \caption{$\quad\fndef[blc]{Commitment}$} \label{alg:commitment}
    \begin{algorithmic}[1]
    \begin{alginfo}
        A bit-level commitment scheme based on a hash function $\fn{H}$. The commitment is implicitly broadcasted after the $\fn{Commit}$ step. Later, the sender reveals the committed value $m$ and the receivers run the $\fn{Open}$ function to verify its validity.
    \end{alginfo}
    \Require {
        $\party{S}: m$, an input message to commit and later open.\\
        $sid$: a unique session identifier (optional, for UC-security).
        }
    \Ensure {
        \textit{valid} if the commitment is verified correctly.
    }
    \vspace{-10pt}
    \AlgFnctZero{\fndef[blc]{Commit}}{}{$m$}{$(c,w)$}
    \State Sample $\sample{w}{\{0,1\}^*}$, a random witness
    \State $\assign{c}{\fn{H}(m\concat w\concat sid)}$, a commitment to $m$
    \Statex \Return $(c,w)$
    \vspace{-12pt}
    \AlgFnct{\fndef[blc]{Open}}{}{$m,c,w$}{\textit{valid}}
    \setalglineno{1}
    \State $\assign{c'}{\fn{H}(m\concat w\concat sid)}$
    \Statex \Return \textit{valid} if $c=c'$, $\ABORT$ otherwise
        
    \end{algorithmic}
\end{scheme}
