\begin{functionality}
    \caption{$\quad\idealfndef{Commitment}$}
    \label{alg:f_commitment}
    \begin{algorithmic}[1]
        \begin{alginfo}
            Commitment scheme, composed of two algorithms:
            \begin{itemize}[leftmargin=12pt]
                \item $\fn{Commit}(m, w)\rightarrow c$: a value $m$ is fixed, a commitment $c$ to that value is generated using a witness $w$ and sent to all receiving parties.
                \item $\fn{Open}(c,m,w)\rightarrow \textit{valid}$ (a.k.a. \textit{reveal}$|$\textit{de-commitment}): committer reveals $m$ alongside $w$, then the receivers validate that it matches the prior commitment $c$.
            \end{itemize}
        \end{alginfo}
    \end{algorithmic}
\end{functionality}
