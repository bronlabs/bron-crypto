\begin{functionality}
    \caption{$\quad\idealfndef{Send}$}\label{func:p2psend}
    \begin{algorithmic}
        \State A functionality for authenticated Peer-To-Peer (P2P) communication between a pair of parties $\party{i}, \party{j}$.
        \Statex \begin{itemize}
            \item On receiving (\textsc{P2PSend-in}$, j, \vect{m}$) from $\party{i}$ with id $i$, $\idealfn{Send}$ sends (\textsc{P2PSend-out}$, i, \vect{m}$) to party $\party{j}$ with id $j$.
        \end{itemize}
    \end{algorithmic}
\end{functionality}
