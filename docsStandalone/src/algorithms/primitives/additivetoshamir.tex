
\begin{algorithm}[H]
    \caption{$\quad\fn{AdditiveToShamir_{t,n,\F}(i, \set{X}, \vecti{x}{i}) \gets \vecti{y}{i}}$} \label{alg:additivetoshamir}
    \begin{algorithmic}[1]
    \Require {
        $i \ins \range{n}$ as the party index \\
        $\set{X} \ins \subrange{t}{n}$ as a subset of $t$ indeces \\
        $\vecti{x}{i} \ins \F$ as a $n$-out-of-$n$ additive share
        }
    \Ensure $\vecti{y}{i} \ins \F$ as the corresponding $t$-out-of-$n$ Shamir share 

    \State Set $\assign{\set{X'}}{ \set{X} \setminus \{i\}}$
    \State $\assign{\ell_i}{\prod_{j \in \set{X'}} \frac{k}{k-i}}$ as the Lagrange coefficient
    \Statex \Return $\assign{\vecti{y}{i}}{\vecti{x}{i} / \ell_i}$ as the Shamir share
    \end{algorithmic}
\end{algorithm}
